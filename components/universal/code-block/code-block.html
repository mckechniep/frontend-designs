Code Block

Styled code display with copy-to-clipboard, optional filename header, line numbers, basic syntax token classes, and inline code variant.

Basic (with copy button)

const createTheme = (name, tokens) => {
  const root = document.documentElement;
  root.dataset.profile = name;

  Object.entries(tokens).forEach(([key, val]) => {
    root.style.setProperty(`--${key}`, val);
  });

  return { name, applied: true };
};
